const validateToken = async (req, res) => {
  try {
    const data = await usersModel.findOne({ secretToken: req.params.token });
    res.send({ data });
  } catch (e) {
    handleHttpError(res, "ERROR_VALIDATE_TOKEN", 500);
  }
};
